2. tạo file env.local
3. tạo tài khoản mongoDB, cloudinary, google_client
4. paste vào file env.local

/n NEXTAUTH_SECRET="ẠDADOAOSDOA"
/n GOOGLE_CLIENT_ID=""
/n GOOGLE_CLIENT_SECRET=""
/n CLOUDINARY_NAME=""
/n CLOUDINARY_KEY=""
/n CLOUDINARY_SECRET=""
/n CLOUDINARY_URL=""
5. npm run dev
